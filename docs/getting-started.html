Interactive Code Examples

Basic Beat-Note Analysis

Learn the fundamentals of chronometric interferometry

basic_example.py
from driftlock_choir import Oscillator, BeatNoteProcessor

# Create two oscillators with slight frequency offset
osc1 = Oscillator.create_ideal_oscillator(2.4e9)  # 2.4 GHz
osc2 = Oscillator.create_ideal_oscillator(2.4e9 + 100)  # +100 Hz

# Generate signals
signal1 = osc1.generate_signal(duration=0.01, sampling_rate=20e6)
signal2 = osc2.generate_signal(duration=0.01, sampling_rate=20e6)

# Process beat note
processor = BeatNoteProcessor(sampling_rate=20e6)
beat_note = processor.generate_beat_note(
    tx_signal=signal1, 
    rx_signal=signal2,
    tx_frequency=2.4e9,
    rx_frequency=2.4e9 + 100
)

# Extract timing information
estimator = EstimatorFactory.create_estimator('phase_slope')
result = estimator.estimate(beat_note)

print(f"Timing precision: {result.tau:.1f} ¬± {result.tau_uncertainty:.1f} ps")
print(f"Frequency accuracy: {result.delta_f:.1f} ¬± {result.delta_f_uncertainty:.1f} Hz")

Two-Node Consensus

Implement distributed timing synchronization

consensus_example.py
from driftlock_choir import MetropolisConsensus, Node

# Create two nodes with different initial frequencies
node1 = Node(frequency=2.4e9, initial_tau=100)  # 100 ps offset
node2 = Node(frequency=2.4e9 + 50, initial_tau=150)  # 150 ps offset

# Set up consensus algorithm
consensus = MetropolisConsensus(
    nodes=[node1, node2],
    convergence_threshold=1e-12,  # 1 ps precision
    max_iterations=100
)

# Run synchronization
result = consensus.synchronize()

print(f"Converged in {result.iterations} iterations")
print(f"Final timing error: {result.final_error:.1f} ps")
print(f"Consensus frequency: {result.consensus_frequency/1e9:.9f} GHz")
